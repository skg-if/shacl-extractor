import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid>
  <Card title="Annotation-driven" icon="pencil">
    Reads property descriptions written directly in `dc:description` annotations on OWL classes. No separate mapping files needed.
  </Card>
  <Card title="Multiple input modes" icon="document">
    Accepts a single ontology file, a modular ontology directory, or a remote URL. Detects the format automatically.
  </Card>
  <Card title="Cardinality to SHACL" icon="list-format">
    Translates cardinality notation like `[1..N]` or `[0..1]` into `sh:minCount` and `sh:maxCount` constraints.
  </Card>
  <Card title="Cross-module references" icon="random">
    When a property points to a class defined in another module, the generated shape references the correct target shape across namespaces.
  </Card>
</CardGrid>
